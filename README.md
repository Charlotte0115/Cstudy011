# Cstudy011
#include <stdio.h>

int main(void)
{
    printf("%d\n", 70);
    printf("%lf\n", 3.1415926535897932384626433);
    printf("%.1lf\n", 3.141592);
    printf("%.10lf\n", 3.141592653589793238462);
    
    printf("%dㅇㅗㅏ %dㅇㅡㅣ ㅎㅏㅂㅇㅡㄴ %dㅇㅣㅂㄴㅣㄷㅏ.\n", 96, 74, 96 + 74);
    printf(%.2lf-%.2lf = %.2lf\n", 5.19, 2.74, 5.19 - 2.74);
    
    return 0;
}
